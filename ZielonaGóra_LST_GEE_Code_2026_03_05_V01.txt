/**
 * Title: Multi-temporal LST and SUHI Analysis of Zielona Góra (2015-2025)
 * Version: 4.2 (Strict Path/Row Centroid Filtering for Exact Image Count)
 */

// --- 1. RESEARCH ASSETS ---
var urbanLULC = ee.FeatureCollection('projects/ee-pinarozguner/assets/UrbanAreas');
var greenLULC = ee.FeatureCollection('projects/ee-pinarozguner/assets/GreenAreas');
var cityBorder = ee.FeatureCollection('projects/ee-pinarozguner/assets/ZGBorders');

var studyArea = cityBorder.geometry().buffer(5000).bounds();
var urbanPolygons = urbanLULC.geometry();
var greenPolygons = greenLULC.geometry();

// --- 2. PRE-PROCESSING & CLOUD MASK ---
function maskClouds(image) {
  var qa = image.select('QA_PIXEL');
  var cloud = qa.bitwiseAnd(1 << 3).neq(0);
  var cirrus = qa.bitwiseAnd(1 << 2).neq(0);
  var shadow = qa.bitwiseAnd(1 << 4).neq(0);
  return image.updateMask(cloud.or(cirrus).or(shadow).not());
}

// --- 3. DYNAMIC LST PHYSICS ENGINE (Sobrino et al., 2004) ---
var applyLST = function(image) {
  var ndvi = image.normalizedDifference(['B5', 'B4']).rename('NDVI');
  var Pv = ndvi.expression('((ndvi - 0.2) / (0.5 - 0.2))**2', {'ndvi': ndvi}).clamp(0, 1);
  var epsilon = Pv.multiply(0.004).add(0.986);
  
  var spacecraft = ee.String(image.get('SPACECRAFT_ID'));
  var lambda = ee.Algorithms.If(spacecraft.equals('LANDSAT_8'), 10.895, 10.885);
  
  var lst = image.select('B10').expression(
    '(BT / (1 + (lambda * BT / 14388) * log(eps))) - 273.15', {
      'BT': image.select('B10'),
      'eps': epsilon,
      'lambda': ee.Number(lambda)
    }).rename('LST');
    
  var cloudCover = image.get('CLOUD_COVER');
  return image.addBands([lst, ndvi]).set('CLOUD_COVER', cloudCover).clip(studyArea);
};

// --- 4. DATASET ASSEMBLY (SUMMER JJA) ---
var collection = ee.ImageCollection("LANDSAT/LC08/C02/T1_TOA")
  .merge(ee.ImageCollection("LANDSAT/LC09/C02/T1_TOA"))
  // FIX: Using the centroid of the city to prevent overlapping adjacent satellite paths
  .filterBounds(cityBorder.geometry().centroid()) 
  .filterDate('2015-01-01', '2025-12-31')
  .filter(ee.Filter.calendarRange(6, 8, 'month'))
  // STAYS THE SAME: Only keeps images with strictly less than 30% cloud cover
  .filter(ee.Filter.lt('CLOUD_COVER', 30))
  .map(maskClouds)
  .map(applyLST);

// --- 5. COMPREHENSIVE STATISTICAL ANALYSIS ---
var years = ee.List.sequence(2015, 2025);

var statsList = years.map(function(y) {
  var yearCol = collection.filter(ee.Filter.calendarRange(y, y, 'year'));
  var count = yearCol.size();
  var meanImg = yearCol.mean();
  
  var avgCloud = ee.Algorithms.If(count.gt(0), yearCol.aggregate_mean('CLOUD_COVER'), 0);
  var sumCloud = ee.Algorithms.If(count.gt(0), yearCol.aggregate_sum('CLOUD_COVER'), 0);
  
  var uTemp = ee.Algorithms.If(count.gt(0), 
    meanImg.select('LST').reduceRegion({
      reducer: ee.Reducer.mean(), geometry: urbanPolygons, scale: 30, maxPixels: 1e9
    }).get('LST'), 0);
    
  var gTemp = ee.Algorithms.If(count.gt(0), 
    meanImg.select('LST').reduceRegion({
      reducer: ee.Reducer.mean(), geometry: greenPolygons, scale: 30, maxPixels: 1e9
    }).get('LST'), 0);
  
  return ee.Feature(null, {
    'Year': ee.String(ee.Number(y).format('%04d')), 
    'Total_Images': count,
    'Mean_Cloud_Percent': avgCloud,
    'Urban_LST': uTemp,
    'Green_LST': gTemp,
    'SUHI_Intensity': ee.Algorithms.If(count.gt(0), ee.Number(uTemp).subtract(gTemp), 0),
    'Sum_Cloud': sumCloud 
  });
});

var finalTable = ee.FeatureCollection(ee.List(statsList));

// --- 6. CALCULATE OVERALL MEANS & PEARSON CORRELATION ---
var totalImagesOverall = collection.size();
var totalCloudSum = finalTable.aggregate_sum('Sum_Cloud');
var weightedMeanCloud = ee.Number(totalCloudSum).divide(totalImagesOverall);

var meanUrbanLST = finalTable.aggregate_mean('Urban_LST');
var meanGreenLST = finalTable.aggregate_mean('Green_LST');
var meanSUHI = finalTable.aggregate_mean('SUHI_Intensity');

var pearsonStats = finalTable.reduceColumns({
  reducer: ee.Reducer.pearsonsCorrelation(),
  selectors: ['Mean_Cloud_Percent', 'Urban_LST']
});

print('--- FINAL DATA SUMMARY ---');
print('Overall Total Images:', totalImagesOverall);
print('Weighted Mean Cloud Cover (%):', weightedMeanCloud);
print('Overall Mean Urban LST (°C):', meanUrbanLST);
print('Overall Mean Green LST (°C):', meanGreenLST);
print('Overall Mean SUHI (°C):', meanSUHI);
print('Pearson Correlation (Cloud vs Urban LST):', pearsonStats.get('correlation'));

// --- 7. MONTHLY IMAGE DISTRIBUTION (JUNE, JULY, AUGUST) ---
var months = ee.List([6, 7, 8]);
var monthNames = ee.Dictionary({6: 'June', 7: 'July', 8: 'August'});

var monthlyCounts = months.map(function(m) {
  var count = collection.filter(ee.Filter.calendarRange(m, m, 'month')).size();
  return ee.Feature(null, {
    'Month': monthNames.get(m), 
    'Image_Count': count,
    'Month_Num': m
  });
});

var monthlyTable = ee.FeatureCollection(monthlyCounts);

print('--- IMAGES PER MONTH ---', monthlyTable.select(['Month', 'Image_Count']));

var monthlyChart = ui.Chart.feature.byFeature(monthlyTable, 'Month', 'Image_Count')

    hAxis: {title: 'Month'},
    vAxis: {title: 'Number of Images', minValue: 0},
    colors: ['#2ca02c']
  });
print(monthlyChart);

var scatterChart = ui.Chart.feature.byFeature(finalTable, 'Mean_Cloud_Percent', 'Urban_LST')
